A path in a binary tree is a sequence of nodes where each pair of adjacent nodes in the sequence has an edge connecting them. A node can only appear in the sequence at most once. Note that the path does not need to pass through the root.

The path sum of a path is the sum of the node's values in the path.

Given the root of a binary tree, return the maximum path sum of any non-empty path.


class Solution {
    int ans = Integer.MIN_VALUE;

    public int maxPathSum(TreeNode root) {
        dfs(root);
        return ans;
    }

    private int dfs(TreeNode root) {
        if (root == null) return 0;

        int left = Math.max(0, dfs(root.left));   // ignore negative
        int right = Math.max(0, dfs(root.right)); // ignore negative

        // case: path THROUGH this node
        ans = Math.max(ans, left + right + root.val);

        // return: best ONE side path
        return root.val + Math.max(left, right);
    }
}
You're returning x(root) instead of ans.
x(root) returns the best single downward branch from root — that's not the answer. The answer is ans, which tracks the best path that can bend at any node.

Bug 1: ans not initialized to Integer.MIN_VALUE
You initialize ans = 0, but the tree can contain all negative values. For a tree like [-3], the answer is -3, but your code returns 0.
Bug 2: Base case return 0 is wrong for the same reason
When a child is null, you return 0 (meaning "take nothing"). That's correct conceptually — you just don't extend the path through null. But since ans starts at 0, a single negative node is never recorded. Fixing ans initialization alone resolves this.
Bug 3 (Logic): Base case calls maxPathSum recursively — wrong method
Your recursive call maxPathSum(root.left) re-enters the public method, which re-checks root == null and returns 0. This accidentally works, but it's semantically wrong — the public method isn't designed to be the recursive helper. It'll also reset ans conceptually (though since ans is an instance field it survives — still, it's fragile and confusing).

class Solution {
    public ListNode mergeKLists(ListNode[] lists) {
        PriorityQueue<ListNode> pq = new PriorityQueue<>((a, b) -> a.val - b.val);

        for (ListNode i : lists) {
            while (i != null) {
                pq.add(i);
                i = i.next;
            }
        }

        ListNode dummy = new ListNode();
        ListNode root = dummy;

        while (!pq.isEmpty()) {
            ListNode node = pq.poll();

            root.next = node;
            root = root.next;

            root.next = null; // 🔥 important to avoid old links
        }

        return dummy.next;
    }
}

lass Solution {
    public ListNode mergeKLists(ListNode[] lists) {
        if (lists == null || lists.length == 0) return null;
        ListNode root  = lists[0];
        for(int i = 1; i < lists.length; i++) {
            root = merge(root, lists[i]);
        }
        return root;
    } 


    ListNode merge(ListNode l1, ListNode l2) {
        if (l1 != null && l2 != null) {
            if (l1.val < l2.val) {
                l1.next = merge(l1.next, l2);
                return l1;
            } else {
                l2.next = merge(l1, l2.next);
                return l2;
            }   
        } else if (l1 == null) return l2;
        else return l1;
    }


}

    }

class Solution {
    
    class TrieNode {
        TrieNode[] children = new TrieNode[26];
        boolean isWord;
    }
    
    TrieNode root = new TrieNode();
    Boolean[] dp;
    
    public boolean wordBreak(String s, List<String> wordDict) {
        buildTrie(wordDict);
        dp = new Boolean[s.length()];
        return solve(0, s);
    }
    
    private boolean solve(int i, String s) {
        if (i == s.length()) return true;
        
        if (dp[i] != null) return dp[i];
        
        TrieNode node = root;
        
        for (int j = i; j < s.length(); j++) {
            char c = s.charAt(j);
            
            if (node.children[c - 'a'] == null) break;
            
            node = node.children[c - 'a'];
            
            if (node.isWord && solve(j + 1, s)) {
                return dp[i] = true;
            }
        }
        
        return dp[i] = false;
    }
    
    private void buildTrie(List<String> dict) {
        for (String word : dict) {
            TrieNode node = root;
            for (char c : word.toCharArray()) {
                if (node.children[c - 'a'] == null) {
                    node.children[c - 'a'] = new TrieNode();
                }
                node = node.children[c - 'a'];
            }
            node.isWord = true;
        }
    }
}

class Solution {
    public Node cloneGraph(Node node) {
        if (node == null) return null;

        Queue<Pair<Node, Node>> q = new LinkedList<>();

        Node cloneRoot = new Node(node.val);

        q.add(new Pair<>(node, cloneRoot));

        Map<Node, Node> map = new HashMap<>();
        map.put(node, cloneRoot);

        while (!q.isEmpty()) {
            Pair<Node, Node> x = q.poll();

            Node original = x.getKey();
            Node clone = x.getValue();

            List<Node> cloneNeighbors = new ArrayList<>();

            for (Node i : original.neighbors) {

                Node j;

                if (map.containsKey(i)) {
                    j = map.get(i);
                } else {
                    j = new Node(i.val);
                    map.put(i, j);
                    q.add(new Pair<>(i, j));
                }

                cloneNeighbors.add(j);
            }

            clone.neighbors = cloneNeighbors;
        }

        return cloneRoot;
    }
}

class Solution {

    Map<Node, Node> map = new HashMap<>();

    public Node cloneGraph(Node node) {
        if (node == null) return null;

        return dfs(node);
    }

    private Node dfs(Node r) {

        if (map.containsKey(r)) {
            return map.get(r);
        }

        Node c = new Node(r.val);

        map.put(r, c);

        List<Node> n = new ArrayList<>();

        for (Node i : r.neighbors) {
            n.add(dfs(i));
        }

        c.neighbors = n;

        return c;
    }
}

Climbing stairs problem

Why your intuition feels right

You are treating:

1 + recursion(...)

as:

"I found one way by taking this step."

But that 1 is added every time you make a jump, not when you successfully reach the top.

In counting problems, the base case should contribute the count:

if (i == n) return 1;

and recursive calls should simply add the counts:

return recursion(i+1, n) + recursion(i+2, n);

Your code is counting edges (jumps) in the recursion tree, not complete paths from start to finish.

The first failing input is already:

n = 2

where your answer is 1 instead of 2.

class Solution {
    public int climbStairs(int n) {
        if (n <= 2) return n;

        int a = 1, b = 2;
        for (int i = 3; i <= n; i++) {
            int c = a + b;
            a = b;
            b = c;
        }
        return b;
    }
}
Time: O(n)
Space: O(1)

class Solution {
    public TreeNode invertTree(TreeNode root) {
        if (null == root) return root;
        root.left = invertTree(root.right);
        root.right = invertTree(root.left);
        return root;
    }
}

whats wrong

The issue is that you're overwriting root.left before using its original value.

Your code:

root.left = invertTree(root.right);
root.right = invertTree(root.left);

After the first line executes:

root.left = invertTree(root.right);

the original root.left is lost.

Then the second line becomes effectively:

root.right = invertTree(<already inverted right subtree>);

So both children end up being derived from the original right subtree.

Example

Original tree:

    1
   / \
  2   3

First line:

root.left = invertTree(3);

Tree becomes:

    1
   / \
  3   3

(Conceptually, left now points to the inverted right subtree.)

You are given a string s and an integer k. You can choose any character of the string and change it to any other uppercase English character. You can perform this operation at most k times.

Return the length of the longest substring containing the same letter you can get after performing the above operations.

 

Example 1:

Input: s = "ABAB", k = 2
Output: 4



class Solution {
    public int characterReplacement(String s, int k) {
        int l = 0, r = 0,  maxi = 1, maxC = 0;
        int c[] = new int[26];
        while (r < s.length()) {
            maxC = Math.max(maxC, ++c[s.charAt(r)-'A']);
            if (r-l+1 - maxC > k) {
                c[s.charAt(l)-'A']--;
                l++;
            }
            maxi = Math.max(maxi, r-l+1);
            r++;
        }
        return maxi;
    }
}

Given a string s, return the number of palindromic substrings in it.

A string is a palindrome when it reads the same backward as forward.

A substring is a contiguous sequence of characters within the string.

 
 
class Solution {
    public int countSubstrings(String s) {
        int ans  = 0;
        Boolean dp[][] = new Boolean[s.length()][s.length()];
        
        for(int i = 0; i < s.length(); i++) {
            for (int j = i+1; j < s.length(); j++) {
                if (s.charAt(i) == s.charAt(j)) {
                    if (solve(i+1, j-1, s, dp)) {
                        ans += 1;
                    }
                }
            }
        }
        return ans + s.length();
    }

    private boolean solve(int i, int j, String s, Boolean  [][] dp) {
        if (i >= j) return true;
        if (i >= s.length() || j < 0) return false;
        if (dp[i][j] != null) return dp[i][j];
        if (s.charAt(i) == s.charAt(j)) {
            return dp[i][j] = solve(i+1, j-1, s, dp);
        } else return false;
    }
}

class Solution {
public:
    int eraseOverlapIntervals(vector<vector<int>>& intervals){
        const int R = 5 * 1e4 + 1;
        int n = intervals.size();
        int M = 0;
        for(int i = 0; i < n; i ++){
            intervals[i][0] += R;
            intervals[i][1] += R;
            M = max(M, intervals[i][1]);
        }
        vector<vector<int> > endingAt(M + 1);
        for(int i = 0; i < n; i ++){
            endingAt[intervals[i][1]].push_back(i);
        }

        vector<int> DP(M + 1, 0);
        for(int t = 1; t <= M; t ++){
            DP[t] = DP[t - 1];
            for(auto i_id : endingAt[t]){
                DP[t] = max(DP[t], 1 + DP[intervals[i_id][0]]);
            }
        }
        return n - DP[M];
    }
};

public class Solution {
    public String longestPalindrome(String s) {
        if (s.length() <= 1) {
            return s;
        }

        int maxLen = 1;
        String maxStr = s.substring(0, 1);
        s = "#" + s.replaceAll("", "#") + "#";
        int[] dp = new int[s.length()];
        int center = 0;
        int right = 0;

        for (int i = 0; i < s.length(); i++) {
            if (i < right) {
                dp[i] = Math.min(right - i, dp[2 * center - i]);
            }

            while (i - dp[i] - 1 >= 0 && i + dp[i] + 1 < s.length() && s.charAt(i - dp[i] - 1) == s.charAt(i + dp[i] + 1)) {
                dp[i]++;
            }

            if (i + dp[i] > right) {
                center = i;
                right = i + dp[i];
            }

            if (dp[i] > maxLen) {
                maxLen = dp[i];
                maxStr = s.substring(i - dp[i], i + dp[i] + 1).replaceAll("#", "");
            }
        }

        return maxStr;
    }
}
