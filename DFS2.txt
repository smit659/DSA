Given a string s, partition s such that every substring of the partition is a palindrome. Return all possible palindrome partitioning of s.

 

Example 1:

Input: s = "aab"
Output: [["a","a","b"],["aa","b"]]


class Solution {
    public List<List<String>> partition(String s) {
        List<List<String>> ans = new ArrayList<>();
        List<String> res = new ArrayList<>();
        solve(ans, 0, s, res);
        return ans;
    }

    private void solve(List<List<String>> ans,int j , String s, List<String> res) {
     
        if (j >= s.length()) {
            ans.add(new ArrayList<>(res)); return;
        }
        
        StringBuilder temp = new StringBuilder();

        for(int i = j; i < s.length(); i++) {
            temp.append(s.charAt(i));
            if (isPalin(temp.toString())) {
                res.add(temp.toString());
                solve(ans, i+1, s, res);
                res.remove(res.size()-1);
            }
        }
    }

    private boolean isPalin(String a) {
        for(int i = 0; i < a.length()/2 ; i++) {
            if (a.charAt(i) != a.charAt(a.length()-1-i))
                return false;
        }
        return true;
    }
}

class Solution {

    public List<List<String>> partition(String s) {
        int n = s.length();
        boolean[][] dp = new boolean[n][n];

        // Precompute palindrome
        for (int len = 1; len <= n; len++) {
            for (int i = 0; i + len - 1 < n; i++) {
                int j = i + len - 1;

                if (s.charAt(i) == s.charAt(j)) {
                    if (len <= 2) dp[i][j] = true;
                    else dp[i][j] = dp[i + 1][j - 1];
                }
            }
        }

        List<List<String>> ans = new ArrayList<>();
        backtrack(0, s, new ArrayList<>(), ans, dp);
        return ans;
    }

    private void backtrack(int start, String s,
                           List<String> curr,
                           List<List<String>> ans,
                           boolean[][] dp) {

        if (start == s.length()) {
            ans.add(new ArrayList<>(curr));
            return;
        }

        for (int end = start; end < s.length(); end++) {
            if (dp[start][end]) {
                curr.add(s.substring(start, end + 1));
                backtrack(end + 1, s, curr, ans, dp);
                curr.remove(curr.size() - 1);
            }
        }
    }
}

import java.util.*;

class MinStack {

    Stack<Integer> stack;
    TreeMap<Integer, Integer> map;

    public MinStack() {
        stack = new Stack<>();
        map = new TreeMap<>();
    }
    
    public void push(int val) {
        stack.push(val);
        map.put(val, map.getOrDefault(val, 0) + 1);
    }
    
    public void pop() {
        int val = stack.pop();

        if (map.get(val) == 1) {
            map.remove(val);
        } else {
            map.put(val, map.get(val) - 1);
        }
    }
    
    public int top() {
        return stack.peek();
    }
    
    public int getMin() {
        return map.firstKey();
    }
}

class MinStack {
	private Node head;
        
    public void push(int x) {
        if (head == null) 
            head = new Node(x, x, null);
        else 
            head = new Node(x, Math.min(x, head.min), head);
    }
    
    public void pop() {
        head = head.next;
    }
    
    public int top() {
        return head.val;
    }
    
    public int getMin() {
        return head.min;
    }
        
    private class Node {
        int val;
        int min;
        Node next;
            
        private Node(int val, int min, Node next) {
            this.val = val;
            this.min = min;
            this.next = next;
        }
    }
}

    public int search(int[] nums, int target) {
        int l = 0 , h = nums.length-1;
        while (l <= h) {
            int m = l + (h-l)/2;

class Solution {
    public int maxSubArray(int[] nums) {
        int sum = 0, maxi = Integer.MIN_VALUE;
        for(int i = 0; i < nums.length; i++) {
            sum += nums[i];
            maxi = Math.max(sum, maxi);
            if (sum < 0) sum = 0;
        }
        return maxi;
    }

public int maxSubArray(int[] A) {
        int n = A.length;
        int[] dp = new int[n];//dp[i] means the maximum subarray ending with A[i];
        dp[0] = A[0];
        int max = dp[0];
        
        for(int i = 1; i < n; i++){
            dp[i] = A[i] + (dp[i - 1] > 0 ? dp[i - 1] : 0);
            max = Math.max(max, dp[i]);
        }
        
        return max;
}

MedianFinder() initializes the MedianFinder object.

class MedianFinder {
    PriorityQueue<Integer> pq1, pq2;
    public MedianFinder() {
        pq1 = new PriorityQueue<>(Collections.reverseOrder());
        pq2 = new PriorityQueue<>();
    }
    
    public void addNum(int num) {
        if (pq1.size() == pq2.size()) {
            pq2.add(num);
            pq1.add(pq2.poll());
        } else {pq2.add(num); pq1.add(pq2.poll()); pq2.add(pq1.poll());}
    }
    
    public double findMedian() {
        if ((pq1.size() + pq2.size()) % 2 !=0){
            return pq1.peek();
        }
        return (pq1.peek() + pq2.peek()) /2.0;
    }
}
