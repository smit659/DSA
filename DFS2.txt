Given a string s, partition s such that every substring of the partition is a palindrome. Return all possible palindrome partitioning of s.

 

Example 1:

Input: s = "aab"
Output: [["a","a","b"],["aa","b"]]


class Solution {
    public List<List<String>> partition(String s) {
        List<List<String>> ans = new ArrayList<>();
        List<String> res = new ArrayList<>();
        solve(ans, 0, s, res);
        return ans;
    }

    private void solve(List<List<String>> ans,int j , String s, List<String> res) {
     
        if (j >= s.length()) {
            ans.add(new ArrayList<>(res)); return;
        }
        
        StringBuilder temp = new StringBuilder();

        for(int i = j; i < s.length(); i++) {
            temp.append(s.charAt(i));
            if (isPalin(temp.toString())) {
                res.add(temp.toString());
                solve(ans, i+1, s, res);
                res.remove(res.size()-1);
            }
        }
    }

    private boolean isPalin(String a) {
        for(int i = 0; i < a.length()/2 ; i++) {
            if (a.charAt(i) != a.charAt(a.length()-1-i))
                return false;
        }
        return true;
    }
}

class Solution {

    public List<List<String>> partition(String s) {
        int n = s.length();
        boolean[][] dp = new boolean[n][n];

        // Precompute palindrome
        for (int len = 1; len <= n; len++) {
            for (int i = 0; i + len - 1 < n; i++) {
                int j = i + len - 1;

                if (s.charAt(i) == s.charAt(j)) {
                    if (len <= 2) dp[i][j] = true;
                    else dp[i][j] = dp[i + 1][j - 1];
                }
            }
        }

        List<List<String>> ans = new ArrayList<>();
        backtrack(0, s, new ArrayList<>(), ans, dp);
        return ans;
    }

    private void backtrack(int start, String s,
                           List<String> curr,
                           List<List<String>> ans,
                           boolean[][] dp) {

        if (start == s.length()) {
            ans.add(new ArrayList<>(curr));
            return;
        }

        for (int end = start; end < s.length(); end++) {
            if (dp[start][end]) {
                curr.add(s.substring(start, end + 1));
                backtrack(end + 1, s, curr, ans, dp);
                curr.remove(curr.size() - 1);
            }
        }
    }
}

You are given an array of non-overlapping intervals intervals where intervals[i] = [starti, endi] represent the start and the end of the ith interval and intervals is sorted in ascending order by starti. You are also given an interval newInterval = [start, end] that represents the start and end of another interval.

Insert newInterval into intervals such that intervals is still sorted in ascending order by starti and intervals still does not have any overlapping intervals (merge overlapping intervals if necessary).

Return intervals after the insertion.

Note that you don't need to modify intervals in-place. You can make a new array and return it.

 

Example 1:

Input: intervals = [[1,3],[6,9]], newInterval = [2,5]
Output: [[1,5],[6,9]]



class Solution {
    public int[][] insert(int[][] intervals, int[] newInterval) {
        List<int[]> l = new ArrayList<>();
        int s = newInterval[0];
        int e = newInterval[1];
        boolean pending = true;
        for(int i = 0; i < intervals.length; i++) {
            int q = intervals[i][0], x = intervals[i][1];
          if (intervals[i][0] <= s && intervals[i][1] >= s) {
            while (i < intervals.length) {
                if (intervals[i][1] >= e && intervals[i][0] <= e) {
                    x = intervals[i][1];
                    break;
                } else if (intervals[i][0] > e) {
                    i--;
                    x = e;
                    break;
                }
                i++;
            }
            if (i == intervals.length)
                x = e;
            l.add(new int[]{q, x});
            pending =  false;
          } else if (intervals[i][0] > s && pending) {
            q = s;
            pending = false;
            while (i < intervals.length) {
                if (intervals[i][1] >= e && intervals[i][0] <= e) {
                    x = intervals[i][1];
                    break;
                } else if (intervals[i][0] > e) {
                    i--;
                    x = e;
                    break;
                }
                i++;
            }
            if (i == intervals.length)
                x = e;
            l.add(new int[]{q, x});
          } else l.add(new int[]{q, x});
        }
        if (pending) {
            l.add(newInterval);
        }
        return l.toArray(new int[l.size()][]);
    }
}

import java.util.*;

class MinStack {

    Stack<Integer> stack;
    TreeMap<Integer, Integer> map;

    public MinStack() {
        stack = new Stack<>();
        map = new TreeMap<>();
    }
    
    public void push(int val) {
        stack.push(val);
        map.put(val, map.getOrDefault(val, 0) + 1);
    }
    
    public void pop() {
        int val = stack.pop();

        if (map.get(val) == 1) {
            map.remove(val);
        } else {
            map.put(val, map.get(val) - 1);
        }
    }
    
    public int top() {
        return stack.peek();
    }
    
    public int getMin() {
        return map.firstKey();
    }
}

BEST Solution
class MinStack {
	private Node head;
        
    public void push(int x) {
        if (head == null) 
            head = new Node(x, x, null);
        else 
            head = new Node(x, Math.min(x, head.min), head);
    }
    
    public void pop() {
        head = head.next;
    }
    
    public int top() {
        return head.val;
    }
    
    public int getMin() {
        return head.min;
    }
        
    private class Node {
        int val;
        int min;
        Node next;
            
        private Node(int val, int min, Node next) {
            this.val = val;
            this.min = min;
            this.next = next;
        }
    }
}
